99  offset, remain)

147  integer, save :: pos = 1, cnt = 0
148  character(len=100), save :: arg0 ! used for option name
149  character(len=512), save :: arg1 ! used for option argument
150 
151  integer :: length, st, id
152  character :: ch
153  logical :: flagLong
154  character*128 lll
155 
156 
157  if (cnt == 0) cnt = command_argument_count()
158 
159  if (pos > cnt) then
160  ! no more arguments left
161  st = 3
162  goto 10
163  end if
164 
165  call get_command_argument(pos, arg0, length)
166  flaglong = .false.
167 
168 
169  !-- This part tries to find the "option id"
170 
171  if (arg0(1:1) == '-') then
172  ! Argument is an option
173 
174  ch = arg0(2:2)
175 
176 
177  if (ch /= '-' .and. len_trim(arg0) > 2) then
178  ! single dash, but name too long ('-xxxx...')
179  write(lll, *) 'Invalid name for short option: "', trim(arg0(2:)), '"'
180  call give_error(lll)
181  end if
182 
183  if (ch == '-' .and. arg0(3:3) == ' ') then
184  ! forced stop ('--')
185  st = 1
186  goto 10
187  end if
188 
189 
190  if (ch == '-') then
191  ! long option ('--xxx...')
192  flaglong = .true.
193  id = lookup_long(arg0(3:))
194  else
195  ! short option
196  id = lookup_short(ch)
197  end if
198 
199  if (id == 0) then
200  write(lll, *) 'Invalid option: "', trim(arg0), '"'
201  call give_error(lll)
202  end if
203 
204  if (.not. options(id)%has_arg) then
205  st = 0
206  goto 10
207  end if
208 
209  if (pos == cnt) then
210  write(lll, *) 'Option "', trim(arg0), '" requires an argument, but none found'
211  call give_error(lll)
212  end if
213 
214  pos = pos + 1
215  call get_command_argument(pos, arg1, length)
216 
217  ! make sure argument is not an option
218  if (arg1(1:1) == '-') then
219  write(lll, *) 'Option "', trim(arg0), '" requires an argument, but option found'
220  call give_error(lll)
221  end if
222 
223  st = 0
224  goto 10
225  end if
226 
227  !-- Not an option, parsing stops
228 
229  st = 2
230  ! we are already at the first non-option argument
231  ! go one step back to the last option or option argument
232  pos = pos - 1
233 
234 
235  !-- Error handling and setting of return values
236  10 continue
237 
238  offset = pos
239  stat = st
240  if (stat == 0) then
241  optindex = id
242  remain = cnt-pos
243  if (options(id)%has_arg) then
244  arg = arg1
245  arglen = length
246  end if
247 
248  ! setup pos for next call to getopt
249  pos = pos + 1
250  end if

298  subroutine print_opt(opt, unit)
299 
300  type(option), intent(in) :: opt
302  integer, intent(in) :: unit
303 
304  integer :: l0, l, c1, c2, idx, new_start
305  character(:), allocatable :: descr_till_end, s_to_wrap, temp
306 
307  if (opt%has_arg) then
308  if (opt%chr .ne. ' ') then ! option may or may not have short version
309  write (unit, '(1x,"-",a,1x,1h<,a,4h> or)') opt%chr, trim(opt%argname)
310  end if
311  ! long version is mandatory
312  write (unit, '(1x,"--",a,2h <,a,1h>)') trim(opt%name), trim(opt%argname)
313  else
314  if (opt%chr .ne. ' ') then
315  write (unit, '(1x,"-",a, 3h or)') opt%chr
316  end if
317  write (unit, '(1x,"--",a)') trim(opt%name)
318  end if
319 
320  if (len_trim(opt%default_) .gt. 0) then
321  write(unit, ' (4x,2h[=,a,1h])') trim(opt%default_)
322  end if
323 
324 
325  ! Splits description using "<br>" as separator, and wraps each piece to print
326  ! at 80 columns maximum
327  descr_till_end = trim(opt%descr)
328  do
329  l0 = len(descr_till_end)
330  if (l0 .eq. 0) exit
331 
332  idx = index(descr_till_end, '<br>')
333 
334  if (idx .eq. 0) then
335  s_to_wrap = descr_till_end
336  else
337  s_to_wrap = descr_till_end(:idx-1)
338  end if
339 
340 
341  ! Wraps string s_to_wrap printing each line
342  ! with 4 characters of indentation at the beginning
343 
344  l = len(s_to_wrap)
345  c1 = 1 ! position of first character of the line to print
346  do
347  if (c1 > l) exit
348 
349  ! prints at maximum 4+76 = 80 characters
350  c2 = min(c1 + 76, l) ! position of the last character of the line to print
351  ! if not at the end of the whole s_to_wrap
352  if (c2 /= l) then
353  ! find the end of a word
354  do
355  if (descr_till_end(c2:c2) == ' ') exit
356  c2 = c2-1
357  end do
358  end if
359  write (unit, '(4x,a)') descr_till_end(c1:c2)
360  c1 = c2+2
361  end do
362 
363  ! Discards part of descr_till_end
364  if (idx .eq. 0) then
365  new_start = l+1
366  else
367  new_start = l+5
368  end if
369  if (new_start .gt. l0) exit
370 
371  temp = descr_till_end(new_start:)
372  descr_till_end = temp
373  end do
374 
375 ! do
376 !
377 ! if (c1 > l) exit
378 !
379 ! idx = index()
380 !
381 !
382 ! ! print at maximum 4+76 = 80 characters
383 ! c2 = min(c1 + 76, MAX_LEN_DESCR)
384 ! ! if not at the end of the whole string
385 ! if (c2 /= MAX_LEN_DESCR) then
386 ! ! find the end of a word
387 ! do
388 ! if (opt%descr(c2:c2) == ' ') exit
389 ! c2 = c2-1
390 ! end do
391 ! end if
392 ! write (unit, '(4x,a)') opt%descr(c1:c2-1)
393 ! c1 = c2+1
394 ! end do
395 
396 
397  end subroutine

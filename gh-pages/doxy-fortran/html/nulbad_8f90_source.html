387 
388  at(0)=0
389  do i = 1,ipptot
390  at(i) = x_pat * i
391  at(-i) = -at(i)
392  if(at(i) .gt. totlarg) go to 40
393  end do
394 
395  bb = 3*aa
396  write(lll,133) bb
397  133 FORMAT(' LE PROFIL GAUSSIEN PAR LEQUEL ON CONVOLE NE PEUT', &
398  ' AVOIR PLUS DE 121 PTS (60 DE CHAQUE COTE DU CENTRE); ', &

400  ' 3 DEMI-LARGEUR DU CENTRE; ', &

402  ' ELARGISSEZ LE PAS DU CALCUL')
403  call pfant_halt(lll)
404

408

410  write(lll,137) max_p_ift, p_ift
411  137 FORMAT(5x,'LA FCTION PAR LAQUELLE VOUS VOULEZ CONVOLER A ', &

413  ' SUR LA FCTION S OBTIENT EN MULTIPLIANT PAR 6 LE NBRE DE PTS SUR LA DEMI LARGEUR')
414  call pfant_halt(lll)
415  end if
416 
417  do i = 1,p_ift
418  p_tfi(i) = at(i-ifd-1)
419  end do
420 
421  z = c7*aa
422

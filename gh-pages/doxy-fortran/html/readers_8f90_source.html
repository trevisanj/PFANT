67  subroutine read_dissoc(path_to_file)
68  integer, parameter :: UNIT_=199
69  integer i, j, k, m, mmaxj
70  character(len=*) :: path_to_file
71  character*2 symbol, symbol_
72  logical flag_found
73 
74  ! Auxiliary temp variables for reading file
75  integer*4 natomm, nelemm
76  dimension natomm(5), nelemm(5)
77 
78  open(unit=unit_,file=path_to_file, status='old')
79 
80  ! row 01
81  read(unit_,'(2i5, 2f10.5, i10)') dissoc_nmetal, dissoc_nimax, dissoc_eps, dissoc_switer
82 
83  ! rows 02 to NMETAL+1: 6-column rows
84  !
85  !
86  !
87  do i = 1, dissoc_nmetal
88  read (unit_, '(a2, 2x, i6, f10.3, 2i5, f10.5)') &
89  symbol_, dissoc_nelemx(i), dissoc_ip(i), &
90  dissoc_ig0(i), dissoc_ig1(i), dissoc_cclog(i)
91 
92  symbol = adjust_atomic_symbol(symbol_)
93 
94  ! makes sure that elements first and second are h and he, respectively,
95  ! because sat4() and die() count on this
96  select case (i)
97  case (1)
98  if (symbol .ne. ' H') then
99  write(lll,*) 'First element must be hydrogen (" H"), not "', symbol_, '"!'
100  call pfant_halt(lll)
101  end if
102  case (2)
103  if (symbol .ne. 'HE') then
104  write(lll,*) 'First element must be helium ("HE"), not "', symbol_, '"!'
105  call pfant_halt(lll)
106  end if
107  end select
108 
109  dissoc_elems(i) = symbol
110 
111  ! spill check
112  if (dissoc_nelemx(i) .gt. max_dissoc_z) then
113  call pfant_halt('read_dissoc(): metal # '//int2str(i)//': nelemxi = '//&
114  int2str(dissoc_nelemx(i))//' over maximum allowed (MAX_DISSOC_Z='//int2str(max_dissoc_z)//')')
115  end if
116 
117  end do
118 
119 
120 
121  ! rows NMETAL+2 till end-of-file
122  ! col 1 -- "name" of molecule
123  ! cols 2-6 -- c(J, 1-5)
124  ! col 7 -- mmax(j) (number of subsequent columns)/2
125  ! cols 8-... -- maximum of 8 columns here.
126  ! pairs (nelem(m), natom(m)), m = 1 to mmax(j)
127  j = 0
128 
129  1010 continue
130  j = j+1
131 
132  read(unit_, '(a3, 5x, e11.5, 4e12.5, i1, 4(i2,i1))') &
133  dissoc_mol(j), &
134  (dissoc_c(j, k), k=1,5), &
135  dissoc_mmax(j), &
136  (nelemm(m), natomm(m), m=1,4)
137 
138  mmaxj = dissoc_mmax(j)
139  if(mmaxj .eq. 0) then
140  ! note: interesting that Fortran accepts a blank line and reads everything blank or zero
141 
142  go to 1014 ! means end-of-file
143  end if
144 
145  ! consistency check:
146  if (mmaxj .gt. 4) then
147  write(lll,*) 'read_dissoc() molecule "', dissoc_mol(j), &
148  '", mmaxj = ', mmaxj, ' cannot be greater than 4!'
149  call pfant_halt(lll)
150  end if
151 
152  ! consistency check
153  do m = 1, mmaxj
154  flag_found = .false.
155  do i = 1, dissoc_nmetal
156  if (nelemm(m) .eq. dissoc_nelemx(i)) then
157  flag_found = .true.
158  exit
159  end if
160  end do
161 
162  if (.not. flag_found) then
163  write(lll,*) 'read_dissoc() molecule "', dissoc_mol(j), &
164  '" atomic number ', nelemm(m), 'not in atoms list above'
165  call pfant_halt(lll)
166  end if
167  end do
168 
169  do m = 1, mmaxj
170  dissoc_nelem(m,j) = nelemm(m)
171  dissoc_natom(m,j) = natomm(m)
172  end do
173 
174  go to 1010
175 
176  1014 dissoc_nmol = j-1
177 
178  write(lll,*) 'Last molecule considered in dissoc file is ', dissoc_mol(dissoc_nmol)
179  call log_debug(lll)
180 
181  close(unit=unit_)
182  flag_read_dissoc = .true.
183  end
184 end

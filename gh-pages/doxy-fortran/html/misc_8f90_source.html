301  INTEGER, INTENT(IN) :: n

304  INTEGER :: i

319  RECURSIVE SUBROUTINE quick_sort_1(left_end, right_end)
320 
321  INTEGER, INTENT(IN) :: left_end, right_end
322 
323  ! Local variables
324  INTEGER :: i, j, itemp
325  REAL*8 :: reference, temp
326  INTEGER, PARAMETER :: max_simple_sort_size = 6
327 
328  IF (right_end < left_end + max_simple_sort_size) THEN
329  ! Use interchange sort for small lists
330  CALL interchange_sort(left_end, right_end)
331 
332  ELSE
333  ! Use partition ("quick") sort
334  reference = list((left_end + right_end)/2)
335  i = left_end - 1; j = right_end + 1
336 
337  DO
338  ! Scan list from left end until element >= reference is found
339  DO
340  i = i + 1
341  IF (list(i) >= reference) EXIT
342  END DO
343  ! Scan list from right end until element <= reference is found
344  DO
345  j = j - 1
346  IF (list(j) <= reference) EXIT
347  END DO
348 
349 
350  IF (i < j) THEN
351  ! Swap two out-of-order elements
352  temp = list(i); list(i) = list(j); list(j) = temp
353  itemp = order(i); order(i) = order(j); order(j) = itemp
354  ELSE IF (i == j) THEN
355  i = i + 1
356  EXIT
357  ELSE
358  EXIT
359  END IF
360  END DO
361 
362  IF (left_end < j) CALL quick_sort_1(left_end, j)
363  IF (i < right_end) CALL quick_sort_1(i, right_end)
364  END IF
365 
366  END SUBROUTINE quick_sort_1

372  SUBROUTINE interchange_sort(left_end, right_end)
373 
374  INTEGER, INTENT(IN) :: left_end, right_end
375 
376  ! Local variables
377  INTEGER :: i, j, itemp
378  REAL*8 :: temp
379 
380  DO i = left_end, right_end - 1
381  DO j = i+1, right_end
382  IF (list(i) > list(j)) THEN
383  temp = list(i)
384  list(i) = list(j)
385  list(j) = temp
386 
387  itemp = order(i)
388  order(i) = order(j)
389  order(j) = itemp

391  END DO
392  END DO

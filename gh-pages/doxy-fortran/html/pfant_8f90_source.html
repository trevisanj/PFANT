291  kp(nelemi) =uiidui(nelemi)*tem25*exp(-ip(nelemi)*t/econst)
292  1060 continue

295  if (t-0.6) 1084, 1072, 1072
296 
297  ! PRELIMINARY VALUE OF PH AT HIGH TEMPERATURES
298  1084 continue
299  pph = sqrt(hkp *(pg/(1.0+heh)+hkp ))-hkp
300  ph = pph**2/hkp
301  go to 1102
302 
303  ! PRELIMINARY VALUE OF PH AT LOW TEMPERATURES
304  1072 continue
305  if (pg/dhh - 0.1) 1073, 1073, 1074
306 
307  1073 continue

309  go to 1102

311 
312  1074 continue
313  ph = 0.5*(sqrt(dhh*(dhh+4.0*pg/(1.0+heh)))-dhh)
314 
315  ! EVALUATION OF THE FICTITIOUS PRESSURES OF HYDROGEN
316  1102 continue
317  u = (1.0+2.0*heh)/dhh
318  q = 1.0+heh
319  r = (2.0+heh)*sqrt(hkp )
320  s = -1.0*pg
321  x = sqrt(ph)
322  iterat = 0
323 
324  1103 continue
325  f = ((u*x**2+q)*x+r)*x+s
326  df = 2.0*(2.0*u*x**2+q)*x+r

328  if (abs((x-xr)/xr)-epsdie) 1105, 1105, 1106
329 
330  1106 continue

332  if (iterat-50) 1104,1104,1107
333 
334  1107 continue

337  e12.5, 5x,'x2=', e12.5, 5x, 'ph=', e12.5)
338  write(lll, 6108) tem,pg,x,xr,ph
339  call log_warning(lll)
340 
341  go to 1105
342 
343  1104 continue
344  x = xr
345

347

353 
357  p(z_h_star) = pph

382  ! !> @note P was being divided by 100 over and over at each j (molecule). This division has been taken out of loop, but is still an issue, since it is unclear *why* this division is being done.
383  ! !> @todo issue ask blb being divided by 100 is still an issue

390  1402 continue

393  do 1030 i =1,dissoc_nmetal
394  nelemi = dissoc_nelemx(i)
395  fx(nelemi) = -fp(nelemi)+p(nelemi)*(1.0 + kp(nelemi)/pe)

397  1030 continue

433

435  nelemi = dissoc_nelemx(i)

437  go to 1043

443  1041 continue

446  do 2001 i=1,dissoc_nmetal
447  nelemi=dissoc_nelemx(i)
448  wa(i)=log10(p(nelemi)+1.0e-30)
449  2001 continue
450

452  wa(imaxp1) = log10(pe+1.0e-30)
453  delta = 0.0

455  nelemi = dissoc_nelemx(i)

458

460

462  delta = delta + abs(z(nelemi) - 1.0)
463 
464  if (dissoc_switer) 2500,2500,2501
465 
466  2501 continue
467  p(nelemi) = (prev(nelemi) + p(nelemi) )*0.5
468  go to 1050
469 
470  2500 continue
471  p(nelemi) = prev(nelemi)
472  1050 continue
473 
474 
475  ! IONIZATION EQUILIBRIUM
476  perev = 0.0
477  do 1061 i=1,dissoc_nmetal
478  nelemi = dissoc_nelemx(i)
479  perev = perev + kp(nelemi)*p(nelemi)
480  1061 continue
481

483  delta = delta + abs((pe-perev)/pe)
484  pe = (perev + pe)*0.5 ! Note that it has an equivalence with the last element of P

487  if (delta - dissoc_eps) 1051,1051,1052
488 
489  1052 continue

492

496  call log_warning(lll)
497 
498  1051 continue
499  end
